InverseNavigationProperty

The System.Reflection.PropertyInfo of the Entity Framework Core inverse navigation, which may or may not exist. Even if it exists, it may not be exposed as a JSON:API relationship.

    public Person Owner { get; set; }
}

    [HasMany] // InverseNavigationProperty: Article.Owner
    public ICollection<Article> Articles { get; set; }
